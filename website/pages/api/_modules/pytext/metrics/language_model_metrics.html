[docs]def compute_language_model_metric(loss_per_word: float) -> LanguageModelMetric:
    try:
        ppl = math.exp(loss_per_word)
    except OverflowError:
        ppl = float("inf")
    return LanguageModelMetric(perplexity_per_word=ppl)
