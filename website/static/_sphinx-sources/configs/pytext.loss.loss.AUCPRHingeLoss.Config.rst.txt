  .. attribute:: precision_range_lower
  
     the lower range of precision values over
     which to compute AUC. Must be nonnegative, `\leq precision_range_upper`,
     and `leq 1.0`.
  
     :type: float
  
  .. attribute:: precision_range_upper
  
     the upper range of precision values over
     which to compute AUC. Must be nonnegative, `\geq precision_range_lower`,
     and `leq 1.0`.
  
     :type: float
  
  .. attribute:: num_classes
  
     number of classes(aka labels)
  
     :type: int
  
  .. attribute:: num_anchors
  
     The number of grid points used to approximate the
     Riemann sum.
  
     :type: int
